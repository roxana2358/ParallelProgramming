# compile
gcc -Wall serial.c -o serial
mpicc -g -Wall mpi.c -o mpi -lm

# execute
1. program name
2. text/packets file
3. patterns file
4. how many patterns to look for
5. output on file

// test 3600 packets, 5 patterns
./serial txt_files/alice_in_wonderland.txt pattern5.txt 5 > outs/outSer.txt
mpiexec -n 10 ./mpi txt_files/alice_in_wonderland.txt pattern5.txt 5 > outs/outMpi.txt
